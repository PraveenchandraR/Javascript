A. const countCharacters = (S) => 
{
   var B=[0,0];
    for(let i in S)

Q.9 Plaindrome Check


A. var Palin_Check = (str) =>
{
	 let N=str.length;
	 let revstr="";
	 for(let i=N-1;i>=0;i--)
	 {
	   revstr+=str[i];
	 }
	 if(revstr==str)
	 {
	   return "True";
	 }
	 
	   return "False";
	 
}
