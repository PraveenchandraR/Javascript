Q8.Second Smallest


    A.const findSndSmallest = (x,y,z) => {
     // check if X is the largest or not 
  if(x>y && x >z)
  {
    // X is the largest 
    if(y>z)
    {
      // y is the second largest
      return y;
    }
    else
    {
      // z is the second largest
      return z;
    }
  }

  // check if Y is the largest or not 
  if(y>x && y >z)
  {
    // Y is the largest 
    if(x>z)
    {
      // x is the second largest
      return x;
    }
    else
    {
      // z is the second largest
      return z;
    }
  }

  // check if z is the largest or not 
  if(z>x && z>y)
  {
    // Z is the largest 
    if(x>y)
    {
      // x is the second largest
      return x;
    }
    else
    {
      // y is the second largest
      return y;
    }
  }
};
